Inital version

using Statistics

struct Experiment
    p::Float64
    n::Int
    theta::Float64
end

function run(ex)
    val = 1.0
    for i in 1:ex.n
        val *= rand() < ex.p ? 1 + ex.theta : 1 - ex.theta
    end
    
    return val
end

Now we can create and run our first experiment

julia> ex = Experiment(0.55, 100, 0.05)
julia> run(ex)
0.5912653691771662
function sim(ex, N = 10_000)
    res = []
    for _ in 1:N
        push!(res, run(ex))
    end
    
    return quantile(res, (0.05, 0.5, 0.95))
end
